 select content_type , count( content_type) as number_of_images FROM access_logs where date>=timestamp'2016-12-29' and date=date_parse(date_format(now() - interval '1' DAY, '%Y-%m-%d'), '%Y-%m-%d') and hourofday=hour(current_timestamp)-1  AND content_type not LIKE '%text%' AND content_type in ('image/vnd.ms-photo', 'image/webp', 'image/png', 'image/gif', 'image/jpg', 'image/jpeg', 'image/png', 'image/gif') and request_url like '%i10c=img.%' group by content_type;
